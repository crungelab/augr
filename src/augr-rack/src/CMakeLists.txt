cmake_minimum_required(VERSION 3.14)

include(${AUGR_CMAKE}/AugrCore.cmake)

file(GLOB RACK_SRC ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
file(GLOB RACK_INCLUDE ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

file(GLOB PIN_SRC ${CMAKE_CURRENT_SOURCE_DIR}/pin/*.cpp)
file(GLOB PIN_INCLUDE ${CMAKE_CURRENT_SOURCE_DIR}/pin/*.h)

file(GLOB CONTROL_SRC ${CMAKE_CURRENT_SOURCE_DIR}/control/*.cpp)
file(GLOB CONTROL_INCLUDE ${CMAKE_CURRENT_SOURCE_DIR}/control/*.h)

file(GLOB MODULE_SRC ${CMAKE_CURRENT_SOURCE_DIR}/module/*.cpp)
file(GLOB MODULE_INCLUDE ${CMAKE_CURRENT_SOURCE_DIR}/module/*.h)

file(GLOB VOICE_SRC ${CMAKE_CURRENT_SOURCE_DIR}/voice/*.cpp)
file(GLOB VOICE_INCLUDE ${CMAKE_CURRENT_SOURCE_DIR}/voice/*.h)

file(GLOB ARCHIVER_SRC ${CMAKE_CURRENT_SOURCE_DIR}/archiver/*.cpp)
file(GLOB ARCHIVER_INCLUDE ${CMAKE_CURRENT_SOURCE_DIR}/archiver/*.h)

target_sources(${THIS}
  PRIVATE
  ${RACK_SRC}
  ${RACK_INCLUDE}
  ${PIN_SRC}
  ${PIN_INCLUDE}
  ${CONTROL_SRC}
  ${CONTROL_INCLUDE}
  ${MODULE_SRC}
  ${MODULE_INCLUDE}
  ${VOICE_SRC}
  ${VOICE_INCLUDE}
  ${ARCHIVER_SRC}
  ${ARCHIVER_INCLUDE}
)
INTERNAL_AUGR_CORE(${THIS})
